1.剑指offer 09.用两个栈实现队列     
思路: 栈的特点先进后出，队列的特点是先进先出，首先定义两个栈`st1`,`st2`，添加元素直接`st1`压栈即可，删除头节点时弹出
`st2`栈顶，然后将`st2`元素压入`st1`中。
```c++
class CQueue {
public:
    stack<int> st1;
    stack<int> st2;
    CQueue() {
    }
    
    void appendTail(int value) {
        st1.push(value);
    }
    
    int deleteHead() {
        while(!st1.empty()){
            st2.push(st1.top());
            st1.pop();
        }
        if(st2.empty())
            return -1;

        int head = st2.top();
        st2.pop();
        while(!st2.empty()){
            st1.push(st2.top());
            st2.pop();
        }
        return head;
    }
};
```

2.**用队列实现栈**
思路: 用一个队列实现栈，主要是在添加元素的过程中，将队列中的元素顺序进行颠倒即可。   
```c++
class MyStack {
public:
    queue<int> q;
    MyStack() {

    }
    
    void push(int x) {
        int size = q.size();
        q.push(x);
        while(size--){
            int tmp = q.front();
            q.pop();
            q.push(tmp);
        }
    }
    
    int pop() {
        int top = q.front();
        q.pop();
        return top;
    }
    
    int top() {
        return q.front();
    }
    
    bool empty() {
        return q.empty();
    }
};
```

4.155.**最小栈**     
思路: 采用一个辅助栈，存储最小值（比栈顶值小就压栈），在弹出元素的时候注意，如果弹出的元素为栈中最小值，
那么辅助栈也弹出该元素。
```c++
class MinStack {
public:
    stack<int> st;
    stack<int> mini;
    MinStack() {

    }
    
    void push(int val) {
        st.push(val);
        if(mini.empty() || mini.top() >= val)
            mini.push(val);
    }
    
    void pop() {
        if(st.empty())
            return ;
        int tmp = st.top();
        st.pop();
        if(mini.top() == tmp)
            mini.pop();
    }
    
    int top() {
        if(st.empty())
            return -1;
        return st.top();
    }
    
    int getMin() {
        if(st.empty())
            return -1;
        return mini.top();
    }
};
```

5.**面试题03.01.三合一**    
思路: 看半天不知道要表达什么意思，直接看答案推导意思，这道题就是一个数组，
分成三段，每段视为一个栈，按照数组下标计算规则来查询，好像没什么实质性内容。
```c++
class TripleInOne {

    vector<int> s;
    int stackSize;
    int spointer[3];
public:
    TripleInOne(int stackSize) {
        s = vector<int>(stackSize*3, 0);
        this->stackSize = stackSize; 
        spointer[0] = 0;
        spointer[1] = stackSize;
        spointer[2] = stackSize*2;
    }
    
    void push(int stackNum, int value) {
        if(spointer[stackNum] < (stackNum+1)*stackSize){
            s[spointer[stackNum]++] = value;
        }
    }
    
    int pop(int stackNum) {
        int res = -1;
        if(spointer[stackNum] > (stackNum)*stackSize){
            res = s[--spointer[stackNum]];
        }
        return res;
    }
    
    int peek(int stackNum) {
        int res = -1;
        if(spointer[stackNum] > (stackNum)*stackSize){
            res = s[spointer[stackNum]-1];
        }
        return res;
    }
    
    bool isEmpty(int stackNum) {
        return spointer[stackNum] == stackNum*stackSize;
    }
};

```

6.20.**有效括号**  
思路: 首先定义一个栈，将三种不同字符左括号进行压栈，然后判断栈是否为空，如果栈为空说明有非法字符出现（直接返回`false`）,
栈不为空，判断栈顶元素是否为右括号并且与上一个压栈字符是不是对应的左括号，最后判定栈是否为空，为空则是有效括号。 
之前的思路是计算左括号和右括号数量是否相等，但是这种思路不正确，例如`"[{]}"`    
```c++
class Solution {
public:
    bool isValid(string s) {
        stack<int> st;
        char t;
        for(auto c : s){
            if(c == '(' || c == '[' || c == '{'){
                st.push(c);
            }else{
                if(st.empty())
                    return false;
                t = st.top(); st.pop();
                if(t == '(' && c != ')')
                    return false;
                if(t == '[' && c != ']')
                    return false;
                if(t == '{' && c != '}')
                    return false;
            }
        }
        return st.empty();
    }
};
```

9.1047.删除字符串中的所有相邻重复项      
思路: 通过栈来记录遍历的字符，如果与前一个字符相同，那么栈弹出栈顶元素，如果不相等就进行压栈，
最后将栈中元素进行连接起来返回即可。   
```c++
class Solution {
public:
// 输入："abbaca"
//输出："ca"
    string removeDuplicates(string s) {
        string res;
        stack<char> st;
        int len = s.size();
        for(int i=len-1; i>=0; i--){
            if(!st.empty()){
                if(st.top() == s[i]){
                    st.pop();
                    continue;
                }
            }
            st.push(s[i]);
        }
        while(!st.empty()){
            char tmp = st.top();
            st.pop();
            res += tmp;
        }
        return res;
    }
};
执行耗时:16 ms,击败了69.29% 的C++用户
```

10.剑指offer 31.栈的压入、弹出序列   
思路: 将压入序列元素依次压入栈中，检查栈顶元素与弹出序列首位是否相等，相等就弹出栈顶元素，
并且弹出序列指针向后移动一位，如果不相等，继续压栈。  
```c++
class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        stack<int> st;
        int len = popped.size();
        int j = 0;
        for(int i=0; i<len; i++){
            st.push(pushed[i]);
            while(!st.empty() && st.top() == popped[j]){
                ++j;
                st.pop();
            }
        }
        return st.empty();
    }
};
```

11.739.每日温度  
思路: 在遍历元素过程中，利用单调栈记录第一个比当前元素大的索引，如果栈顶元素小于当前元素，那么弹出栈顶元素
并且当前元素为第一个比栈顶大的元素，如果栈顶元素大于当前元素，那么就继续压栈，保持整个栈为单调递减。  
```c++
class Solution {
public:
//输入: temperatures = [30,40,50,60]
//输出:[1,1,1,0]
    vector<int> dailyTemperatures(vector<int>& t) {
        int n = t.size();
        vector<int> res(n,0);
        stack<int> st;

        for(int i=0; i<n; i++){
            while(!st.empty() && t[i]>t[st.top()]){
                auto index = st.top();
                st.pop();
                res[index] = i - index;
            }
            st.push(i);
        }
        return res;
    }
};
```

12.42.接雨水  
思路: 在遍历高度过程中，首先得到左边最大值和右边最大值之间的较小值，然后计算该值与高度值之差，       
最后将这些差值进行累加得到结果。  
```c++
class Solution {
public:
    int trap(vector<int>& height) {
        int res = 0;
        int n = height.size();
        int left[n], right[n];
        memset(left, 0, n*sizeof(int));
        memset(right, 0, n*sizeof(int));
        // left max
        for(int i=1; i<n; i++){
            left[i] = max(left[i-1], height[i-1]);
        }
        // right max
        for(int i=n-1-1; i>=0; i--){
            right[i] = max(right[i+1], height[i+1]);
        }
        for(int i=0; i<n; i++){
            res += max(0, min(left[i], right[i])-height[i]);
        }
        return res;
    }
};
执行耗时:16 ms,击败了31.77% 的C++用户
```

13.84.柱状图中最大的矩形 （单调栈）  
思路: 首先统计每个柱子左边第一个小于当前柱子的下标，然后统计每个柱子右边第一个小于当前柱子的下标，
最后统计能够围成的最大矩形面积。   
```c++
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        int res = 0;
        int left[n], right[n];
        memset(left, 0, n*sizeof(int));
        memset(right, 0, n*sizeof(int));

        // 记录每个柱子 左边第一个小于当前柱子的下标
        left[0] = -1;
        for(int i=1; i<n; i++){
            int t = i-1;
            while(t>=0 && heights[t]>=heights[i]){
                t = left[t];
            }
            left[i] = t;
        }
        // 记录每个柱子 右边第一个小于当前柱子的下标
        right[n-1] = n;
        for(int i=n-1-1; i>=0; i--){
            int t = i+1;
            while(t<n && heights[t]>=heights[i]){
                t = right[t];
            }
            right[i] = t;
        }
        // 求和
        for(int i=0; i<n; i++){
            int sum = heights[i] * (right[i] - left[i] - 1);
            res = max(res, sum);
        }
        return res;
    }

};
```
