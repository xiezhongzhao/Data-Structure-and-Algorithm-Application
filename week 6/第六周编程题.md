class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        if(root == nullptr)
            return ans;

        stack<TreeNode*> st, st1;
        bool flag = true;
        int n = 0;
        st.push(root);

        while(!st.empty() || !st1.empty()){
            vector<int> layer;
            n = flag ? st.size() : st1.size();
            for(int i=0; i<n; i++){
                if(flag){
                    TreeNode* tmp = st.top();
                    st.pop();
                    layer.push_back(tmp->val);

                    if(tmp->left != nullptr)
                        st1.push(tmp->left);
                    if(tmp->right != nullptr)
                        st1.push(tmp->right);
                }

                if(!flag){
                    TreeNode* tmp = st1.top();
                    st1.pop();
                    layer.push_back(tmp->val);

                    if(tmp->right != nullptr)
                        st.push(tmp->right);
                    if(tmp->left != nullptr)
                        st.push(tmp->left);
                }
            }
            flag = !flag;
            ans.push_back(layer);
        }
        return ans;
    }
};
