1.203.移除链表元素      
思路: 首先设置一个dummy节点指向头节点，然后在遍历链表节点pNode时，记录前指针pPre和后指针pNext，
如果节点的值和目标值相等时，调整前指针指向后指针，不相等时当前节点变为前指针。
```c++
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        if(head == nullptr)
            return nullptr;
        ListNode* dummy = new ListNode(-1);
        dummy->next = head;
        ListNode* pNode = head;
        ListNode* pPre = dummy;

        while(pNode){
            ListNode* pNext = pNode->next;
            if(pNode->val == val){
                pPre->next = pNext;
            }else{
                pPre = pNode;
            }
            pNode = pNode->next;
        }

        return dummy->next;
    }
};
```

2.876.链表的中间结点    
思路: 首先遍历一遍链表，找出中间节点所在位置`mid`，然后再次遍历到`mid`位置，返回该节点。 
```c++
class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        if(head == nullptr)
            return nullptr;
        ListNode* dummy = new ListNode(-1);
        dummy->next = head;
        int count = 0;
        ListNode* pNode = head;
        while(pNode){
            ++count;
            pNode = pNode->next;
        }
        ListNode* pMid = head;
        int mid = count/2 + 1;
        while(--mid){
            pMid = pMid->next;
        }
        return pMid;
    }
};
```

3.**删除排序链表中的重复元素**          
思路: 首先定义一个虚拟节点指向头指针，然后开始遍历节点，当前节点`pNode`等于下一个节点`pNode->next`时，
那么当前节点`pNode`指向`pNode->next->next`。
```c++
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head == nullptr || head->next == nullptr)
            return head;
        ListNode* dummy = new ListNode(-1);
        dummy->next = head;

        ListNode* pNode = head;
        while(pNode->next != nullptr){
            if(pNode->val == pNode->next->val)
                pNode->next = pNode->next->next;
            else{
                pNode = pNode->next;
            }
        }
        return dummy->next;
    }
};
```

4.剑指offer 25.合并两个排序的链表     
思路: 首先定义合并后的头指针`pMerge`，然后开始遍历`l1`, `l2`, 比较`l1->val`和`l2->val`的值，
`pMerge`合并较小的值，最后返回合并完成后的链表头节点。  
```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if(l1 == nullptr)
            return l2;
        if(l2 == nullptr)
            return l1;
        ListNode* dummy = new ListNode(-1);
        ListNode* pMerge = new ListNode(0);
        dummy->next = pMerge;

        while(l1 && l2){
            if(l1->val > l2->val){
                pMerge->next = new ListNode(l2->val);
                l2 = l2->next;
            }else if(l1->val < l2->val){
                pMerge->next = new ListNode(l1->val);
                l1 = l1->next;
            }else{
                pMerge->next = new ListNode(l1->val);
                pMerge = pMerge->next;
                pMerge->next = new ListNode(l2->val);

                l1 = l1->next;
                l2 = l2->next;
            }
            pMerge = pMerge->next;
        }
        while(l1){
            pMerge->next = new ListNode(l1->val);
            pMerge = pMerge->next;
            l1 = l1->next;
        }
        while(l2){
            pMerge->next = new ListNode(l2->val);
            pMerge = pMerge->next;
            l2 = l2->next;
        }
        return dummy->next->next;
    }
};
```

5.两数相加      
思路: 首先同步遍历两个链表，两个元素相加（满10进位），然后将相加的元素连接在一起，最后返回相加后链表的头节点。
```c++
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* dummy = new ListNode(-1);
        ListNode* pAdd = new ListNode(0);
        dummy->next = pAdd;

        ListNode* pNode1 = l1;
        ListNode* pNode2 = l2;
        int remainder = 0;
        while(pNode1 && pNode2) {
            int sum = pNode1->val + pNode2->val;
            pAdd->next = new ListNode((sum+remainder)%10);
            if((sum+remainder) >= 10){
                remainder = 1;
            }else{
                remainder = 0;
            }
            pAdd = pAdd->next;
            pNode1 = pNode1->next;
            pNode2 = pNode2->next;
        }
        while(pNode1){
            pAdd->next = new ListNode((pNode1->val + remainder)%10);
            if((pNode1->val + remainder) >= 10){
                remainder = 1;
            }else{
                remainder = 0;
            }
            pAdd = pAdd->next;
            pNode1 = pNode1->next;
        }
        while(pNode2){
            pAdd->next = new ListNode((pNode2->val + remainder)%10);
            if((pNode2->val + remainder) >= 10){
                remainder = 1;
            }else{
                remainder = 0;
            }
            pAdd = pAdd->next;
            pNode2 = pNode2->next;
        }

        if(remainder == 1){
            pAdd->next = new ListNode(1);
            pAdd->next->next = nullptr;
        }else{
            pAdd->next = nullptr;
        }

        return dummy->next->next;
    }
};
```

6.206.反转链表       
思路: 首先定义边界条件（头节点为空或者只有一个节点），然后定义前置节点`pPre`和当前节点`pNode`, 
在遍历链表过程中将当前节点指向前置节点`pNode->next = pPre`，最后将哑节点连接到最后一个节点，
返回`dummy->next`。 
```c++
class Solution {
public:
    //输入：head = [1,2,3,4,5]
    //输出：[5,4,3,2,1]
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr || head->next == nullptr)
            return head;

        ListNode* dummy = new ListNode(-1);
        ListNode* pPre = nullptr;
        ListNode* pNode = head;
        while(pNode){
            ListNode* pNext = pNode->next;
            pNode->next = pPre;
            pPre = pNode;
            pNode = pNext;
        }
        dummy->next = pPre;
        return dummy->next;
    }
};
```

7.234.回文链表      
思路: 首先定义一个栈将链表元素压入栈中，然后弹出栈顶与链表元素进行对比，只有有一个不相等， 直接返回`false`, 
如果元素都相等，返回`true`。
```c++
class Solution {
public:
    bool isPalindrome(ListNode* head) {

        stack<int> st;
        ListNode* pNode = head;
        ListNode* pNode1 = head;
        while(pNode){
            st.push(pNode->val);
            pNode = pNode->next;
        }

        while(!st.empty()){
            if(st.top() != pNode1->val)
                return false;
            st.pop();
            pNode1 = pNode1->next;
        }
        return true;
    }
};
```

返回奇链表的头节点。
```c++
class Solution {
public:
    //输入: head = [1,2,3,4,5]
    //输出:[1,3,5,2,4]
    ListNode* oddEvenList(ListNode* head) {
        if(head == nullptr)
            return nullptr;
        if(head->next == nullptr)
            return head;

        ListNode* dummyOdd = new ListNode(0);
        ListNode* pOdd= new ListNode(-1);
        dummyOdd->next = pOdd;

        ListNode* pEven = new ListNode(-2);
        ListNode* dummyEven= new ListNode(-4);
        dummyEven->next = pEven;

        ListNode* pNode = head;
        int count = 1;
        while(pNode){
            if(count%2 !=0 ){
                pOdd->next = pNode;
                pOdd = pOdd->next;
            }else{
                pEven->next = pNode;
                pEven = pEven->next;
            }
            pNode = pNode->next;
            ++count;
        }

        pEven->next = nullptr;
```

9.25.K个一组翻转链表      
思路: 在遍历链表过程中，以`k`步保存链表元素到容器中，然后对容器中的`k`个数进行反转，最后通过遍历容器元素并且新建该元素节点进行连接。
```c++
class Solution {
public:

//输入：head = [1,2,3,4,5], k = 3
//输出：[3,2,1,4,5]
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(head == nullptr)
            return head;
        ListNode* dummy = new ListNode(-1);
        ListNode* pReversed = dummy;

        ListNode* pNode = head;
        vector<vector<int>> eles;
        vector<int> tmp;
        int count = 1;
        while(pNode){
            tmp.push_back(pNode->val);
            if(count%k == 0){
                reverse(tmp.begin(), tmp.end());
                eles.push_back(tmp);
                tmp.clear();
            }
            pNode = pNode->next;
            ++count;
        }
        for(auto arr : eles){
            for(auto it : arr){
                pReversed->next = new ListNode(it);
                pReversed = pReversed->next;
            }
        }

        for(auto it : tmp){
            pReversed->next = new ListNode(it);
            pReversed = pReversed->next;
        }
        pReversed->next = nullptr;
        return dummy->next;
    }
};
```
