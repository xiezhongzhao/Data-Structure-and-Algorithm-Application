```c++
class Solution {
public:
    int sum(int index){
        int s = 0;
        while(index){
            s += index%10;
            index /= 10;
        }
        return s;
    }

    int dfs(int r, int c, int m, int n, int k, vector<vector<bool>>& used){
        if(r<0|| r>=m || c>=n || c<0 || sum(r)+sum(c) > k || used[r][c] == true)
            return 0;
        used[r][c] = true;
        return 1+dfs(r+1, c, m, n, k, used)+dfs(r-1, c, m, n, k, used)+
               dfs(r, c+1, m, n, k, used)+dfs(r, c-1, m, n, k, used);
    }

    int movingCount(int m, int n, int k) {
        if(m<=0 || n<=0 || k<=0)
            return 0;

        vector<vector<bool>> used(m, vector<bool>(n,0));
        int count = dfs(0, 0, m, n, k, used);
        return count;
    }
};
```

```c++
        return image;
    }
};
```

```c++
class Solution {

        if(row<0||row>=m||col<0||col>=n||grid[row][col]=='0')
            return ;
        grid[row][col] = '0';

                    dfs(grid, rows, cols, i, j);
                    ++count;
                }
            }
        }
        return count;
    }
};
```

```c++
            return 0;
        }

                dfs(land, m, n, row-1, col+1) + dfs(land, m, n, row+1, col-1);

        return res;
    }

    vector<int> pondSizes(vector<vector<int>>& land) {
        vector<int> res;
        int rows = land.size();
        int cols = land[0].size();
        for(int i=0; i<rows; i++){
            for(int j=0; j<cols; j++){
                if(land[i][j] == 0){
                    int num = dfs(land, rows, cols, i, j);
                    res.push_back(num);
                }
            }
        }
        sort(res.begin(), res.end());
        return res;
    }
};
```

```c++
class Solution {
public:
    bool flag = false;
    void dfs(vector<vector<char>>& board, int row, int col, int index,
            string& word, vector<vector<bool>>& used){
        int m = board.size();
        int n = board[0].size();
        if(row<0 || row>=m || col<0 || col>=n || word[index] != board[row][col] || used[row][col] == true)
            return ;
        if(index == word.size()-1)
            flag = true;

        used[row][col] = true;
        dfs(board, row+1, col, index+1, word, used);
        dfs(board, row-1, col, index+1, word, used);
        dfs(board, row, col+1, index+1, word, used);
        dfs(board, row, col-1, index+1, word, used);
        used[row][col] = false;
    }

    bool exist(vector<vector<char>>& board, string word) {
        int rows = board.size();
        int cols = board[0].size();
        vector<vector<bool>> used(rows, vector<bool>(cols,false));
        for(int i=0; i<rows; i++){
            for(int j=0; j<cols; j++){
                dfs(board, i, j, 0, word, used);
                if(flag)
                    return true;
            }
        }
        return flag;
    }
};
```

```c++
class Solution {
public:
    bool flag = false;
    void dfs(vector<int>& arr, int start, vector<bool>& used){
       if(start<0 || start>=arr.size() || used[start]==true)
           return ;
       if(arr[start] == 0){
           flag = true;
           return ;
       }

       used[start] = true;
       dfs(arr, start+arr[start], used);
       dfs(arr, start-arr[start], used);
    }

    bool canReach(vector<int>& arr, int start){
        if(start >= arr.size())
            return flag;

        vector<bool> used(arr.size(), false);
        dfs(arr, start, used);
        return flag;
    }
};
```

10.面试题 17.22. **单词转换**   
思路:
```c++
class Solution {
public:
    vector<string> res;
    unordered_set<string> us;
    bool dfs(string& s, string& ew, int n){
        if(s == ew)
            return true;
        for(int i=0; i<n; i++){
            char cl = s[i];
            for(char c='a'; c<='z'; c++){
                s[i] = c;
                if(us.count(s)){
                    us.erase(s);
                    res.push_back(s);
                    if(dfs(s, ew, n))
                        return true;
                    res.pop_back();
                }
            }
            s[i] = cl;
        }
        return false;
    }

    vector<string> findLadders(string beginWord, string endWord, vector<string>& wordList) {
        us.clear();
        res.clear();
        res.push_back(beginWord);
        for(auto& e : wordList){
            us.insert(e);
        }
        us.erase(beginWord);
        if(us.count(endWord)){
            if(dfs(beginWord, endWord, endWord.size()))
                return res;
        }
        return {};
    }
};
```
